```.py
# While loop
# For given integer n print all squares of integers where the square is less than or equal to n, in ascending order.
n = int(input("Input n: "))
num = 1
sqr = 1
while sqr <= n:
    print(sqr)
    num += 1
    sqr = num ** 2
```
```.py
# For loop
n = int(input())
for i in range(1, n+1):
    if i**2 <= n:
        print(i**2)
```
