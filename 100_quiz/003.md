# Given an integer N, show 100 integers with the repeating pattern 0 to N-1, 0 to N-1,... and output the sum of these integers
def rangePatternN(n):
    output = []
    sum = 0
    for i in range(100):
        for i in range(0, n):
            if len(output) == 100:
                break
            output.append(i)
    for num in output:
        sum += num
    return output, sum
